def css_mtime_for(prefix=nil, fname)
  if settings.css_aggressive_mtime
    prefix ||= File.basename(fname)
    Dir[prefix, "**/*"].map { |f| File.mtime(f).to_i }.max
  else
    File.mtime(fname).to_i
  end
end
